00085 // Perform a 10-bit conversion

00088 {

00091         sbi(ADCSR, ADIF);                                               // clear hardware "conversion complete" flag 
00092         sbi(ADCSR, ADSC);                                               // start conversion
00093         //while(!a2dCompleteFlag);                              // wait until conversion complete
